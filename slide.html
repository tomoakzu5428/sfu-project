            <!-- ドットはJSで生成されます --></div>

    </div>


    <script>

        document.addEventListener('DOMContentLoaded', function() {

            // Lucide Iconsの初期化

            lucide.createIcons();


            const slider = document.getElementById('slider');

            const slides = document.querySelectorAll('.slide');

            const prevBtn = document.getElementById('prevBtn');

            const nextBtn = document.getElementById('nextBtn');

            const pagination = document.getElementById('pagination');

            const totalSlides = slides.length;

            let currentSlide = 0;


            // ページネーションドットの生成

            for (let i = 0; i < totalSlides; i++) {

                const dot = document.createElement('button');

                dot.classList.add('w-3', 'h-3', 'rounded-full', 'transition-all', 'duration-300', 'cursor-pointer');

                dot.classList.add(i === 0 ? 'bg-tc-accent' : 'bg-gray-400', 'hover:bg-tc-accent');

                dot.dataset.slide = i;

                pagination.appendChild(dot);

            }


            const dots = document.querySelectorAll('#pagination button');


            // スライド表示関数

            function showSlide(slideIndex) {

                // インデックスの境界チェック

                if (slideIndex >= totalSlides) {

                    slideIndex = 0;

                } else if (slideIndex < 0) {

                    slideIndex = totalSlides - 1;

                }


                // スライダーの移動

                slider.style.transform = `translateX(-${slideIndex * 100}%)`;

                currentSlide = slideIndex;


                // ページネーションの更新

                updatePagination();

            }


            // ページネーションのドットのアクティブ状態を更新

            function updatePagination() {

                dots.forEach((dot, index) => {

                    if (index === currentSlide) {

                        dot.classList.remove('bg-gray-400');

                        dot.classList.add('bg-tc-accent', 'w-5'); // アクティブなドットを少し大きくする

                    } else {

                        dot.classList.add('bg-gray-400');

                        dot.classList.remove('bg-tc-accent', 'w-5');

                        dot.classList.add('w-3');

                    }

                });

            }


            // イベントリスナー

            nextBtn.addEventListener('click', () => {

                showSlide(currentSlide + 1);

            });


            prevBtn.addEventListener('click', () => {

                showSlide(currentSlide - 1);

            });


            dots.forEach(dot => {

                dot.addEventListener('click', (e) => {

                    showSlide(parseInt(e.target.dataset.slide));

                });

            });


            // キーボード操作（左右矢印キー）

            document.addEventListener('keydown', (e) => {

                if (e.key === 'ArrowRight') {

                    showSlide(currentSlide + 1);

                } else if (e.key === 'ArrowLeft') {

                    showSlide(currentSlide - 1);

                }

            });


            // 初期スライドの表示

            showSlide(currentSlide);

        });

    </script>
